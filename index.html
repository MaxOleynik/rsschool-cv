Code example

                    Find the unique number:
                    
                        function findUniq(arr) {
                            const set = new Set(arr);
                        
                            for(let item of set){
                              let count = 0;
                              for(let num of arr){
                                if(item === num){
                                    count++;
                                }
                              }
                              if(count === 1){
                                  return item;
                              }
                            }
                        }
                    

                    Factorial recursion:
                    
                        function factorial(n){
                            if(n === 0){
                                return 1;
                            }
                        
                            return n * factorial(n - 1);
                        }
